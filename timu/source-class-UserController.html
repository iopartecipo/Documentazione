 42   public function actionLogin() {
 43     if (userIsLogged()) {
 44       $this->redirect(BASE_URL);
 45     }
 46     $response = array();
 47     $backUrl = BASE_URL;
 48     try {
 49       if (isModuleExist('backendconnector') == false) {
 50         throw new Exception(Yii::t('story', 'backendconnector module is missing'));
 51       }
 52       $module = Yii::app()->getModule('backendconnector');
 53       if (empty($module)) {
 54         throw new Exception(Yii::t('story', 'backendconnector module is missing or not defined'));
 55       }
 56       $user = new UserIdentityManager();
 57       if (!empty($_POST)) {
 58         $userDetail = $_POST;
 59         if (empty($userDetail['email']) || !filter_var($userDetail['email'], FILTER_VALIDATE_EMAIL)) {
 60           throw new Exception(Yii::t('story', 'Please enter a valid email'));
 61         }
 62         if (empty($userDetail['password'])) {
 63           throw new Exception(Yii::t('story', 'Please enter password'));
 64         }
 65         $response = $user->validateUser($userDetail);
 66         if (array_key_exists('success', $response) && !empty($response['success'])) {
 67           $isAdmin = checkPermission('is_admin');
 68           $_SESSION['user']['admin'] = $isAdmin;
 69           if (!empty($userDetail['remember_me'])) {
 70             if (isset($response['_items'][0]) && array_key_exists('_id', $response['_items'][0]) && $response['_items'][0]['_id'] !== NULL) {
 71               $encryptedEmail = sha1($response['_items'][0]['email'] . ENCRIPT_SALT);
 72               $cookieUser = CookieUser::model()->find('user_email = :user_email', array(':user_email' => $encryptedEmail));
 73               $token = generateToken();
 74               $encryptedToken = sha1($token . ENCRIPT_SALT);
 75               $sessionTimeOut = 60*60*24;
 76               if (defined('SESSION_TIMEOUT_TIME')) {
 77                 $sessionTimeOut = SESSION_TIMEOUT_TIME;
 78               }
 79               $cookieDomain = '';
 80               if (defined('COOKIE_DOMAIN')) {
 81                 $cookieDomain = COOKIE_DOMAIN;
 82               }
 83               $userCookie = new CHttpCookie('user_email', $encryptedEmail,
 84                 array('expire' => (time() + $sessionTimeOut), 'path' => '/', 'domain' => $cookieDomain));
 85               Yii::app()->request->cookies['user_email'] = $userCookie;
 86               $userToken = new CHttpCookie('user_token', $encryptedToken,
 87                 array('expire' => (time() + $sessionTimeOut), 'path' => '/','domain' => $cookieDomain));
 88               Yii::app()->request->cookies['user_token'] = $userToken;
 89               if (isset($cookieUser->id) && $cookieUser->id != NULL) {
 90                 $cookieUser->user_token = $encryptedToken;
 91                 $updateCookieInfo = $cookieUser->update();
 92                 if($updateCookieInfo == FALSE) {
 93                   Yii::log('', ERROR, 'Error in updating records in cookie table.');
 94                 }
 95               } else {
 96                 $saveInfo = new CookieUser();
 97                 $saveInfo->user_email = $encryptedEmail;
 98                 $saveInfo->user_token = $encryptedToken;
 99                 $saveInfo->user_id = $response['_items'][0]['_id'];
100                 $saveCookieInfo = $saveInfo->save();
101                 if ($saveCookieInfo == FALSE) {
102                   Yii::log('', ERROR, 'Error in saving records in cookie table.');
103                 }
104               }
105             }
106           }
107         }
108         if (!empty($_GET['back'])) {
109           $backUrl = BASE_URL . substr($_GET['back'], 1);
110         }
111       }
112     } catch (Exception $e) {
113       $response['success'] = false;
114       $response['msg'] = $e->getMessage();
115       Yii::log('', ERROR, Yii::t('story', 'Error in actionLogin method :') . $e->getMessage());
116     }
117     $this->layout = 'userManager';
118     $this->render('login', array('message' => $response, 'back_url' => $backUrl));
119   }

172           if ($user['password'] !== $user['confirm_password']) {
173             throw new Exception(Yii::t('story', 'Password does not match'));
174           }
175           if (!array_key_exists('terms_and_condition', $user)) {
176             throw new Exception(Yii::t('story', 'Please check term and condition checkbox'));

179             throw new Exception(Yii::t('story', 'Please check privacy policy checkbox'));
180           }

198             throw new Exception(Yii::t('story', 'backendconnector module is missing or not defined'));

200           $userIdentityMgr = new UserIdentityManager();
201           $saveUser = $userIdentityMgr->createUser($userDetail);
202           if (array_key_exists('success', $saveUser) && $saveUser['success'] == true) {
203             $this->_sendActivationMail($user);
204             $saveUser['msg'] = Yii::t('story', 'You have been successfully registered');

213       $saveUser['msg'] = $e->getMessage();

217     $js = Yii::app()->getClientScript();
218     $js->registerScriptFile(THEME_URL . 'js/userRegistration.js', CClientScript::POS_END);
219     $serviceName = Yii::app()->request->getQuery('service');
220 
221         if (isset($serviceName)) {
222             /** @var $eauth EAuthServiceBase */
223             $eauth = Yii::app()->eauth->getIdentity($serviceName);
224             $eauth->redirectUrl = Yii::app()->user->returnUrl;
225             $eauth->cancelUrl = $this->createAbsoluteUrl('site/login');
226 
227             try {
228                 if ($eauth->authenticate()) {
229                     //var_dump($eauth->getIsAuthenticated(), $eauth->getAttributes());
230                     $identity = new EAuthUserIdentity($eauth);
231 
232                     // successful authentication
233                     if ($identity->authenticate()) {
234                         Yii::app()->user->login($identity);
235                         //var_dump($identity->id, $identity->name, Yii::app()->user->id);exit;
236 
237                         // special redirect with closing popup window
238                         $eauth->redirect();
239                     }
240                     else {
241                         // close popup window and redirect to cancelUrl
242                         $eauth->cancel();
243                     }
244                 }
245 
246                 // Something went wrong, redirect to login page
247                 $this->redirect(array('site/login'));
248             }
249             catch (EAuthException $e) {
250                 // save authentication error to session
251                 Yii::app()->user->setFlash('error', 'EAuthException: '.$e->getMessage());
252 
253                 // close popup window and redirect to cancelUrl
254                 $eauth->redirect($eauth->getCancelUrl());
255             }
256         }
257

259 
260     $this->render('registration', array('message' => $saveUser, 'back_url' => $backUrl, 'user' => $user));
261   }

303    * _prepareMailBody
304    * This funtion is used to create email body for activation mail
305    * @param  array $userInfo - data for mail
306    * @return string $html
307    */
308   private function _prepareMailBody($userInfo) {
309     $html = '';
310     $html = file_get_contents(Yii::app()->theme->basePath . '/views/user/activationEmail.html');
311     $html = str_replace("{{salute_text}}", $userInfo['salute_text'], $html);
312     $html = str_replace("{{user_name}}", $userInfo['firstname'] . ' ' . $userInfo['lastname'] , $html);
313     $mailText = Yii::t('story', 'Thank you for registering. Please activate your account by clicking this {start_ahref_link} link {end_ahref_link}  or copy and paste the link below and follow the instructions.',
314        array('{start_ahref_link}' => '<a href="' . $userInfo['activation_link'] . '" target="_blank">', '{end_ahref_link}' => '</a>' ));
315     $html = str_replace("{{mail_text_description}}", $mailText, $html);
316     $html = str_replace("{{activation_link}}", $userInfo['activation_link'], $html);
317     $html = str_replace("{{regards}}", $userInfo['regards'], $html);
318     return $html;

325   public function actionActivateUser() {
326     try {
327       Yii::app()->clientScript->registerCssFile(THEME_URL . 'css/bootstrap.css');
328       $this->setHeader('3.0');
329       $message = '';
330       $verified = FALSE;
331       if (array_key_exists('u1', $_GET) && array_key_exists('u2', $_GET) && array_key_exists('u3', $_GET)) {
332         $secret_key = $_GET['u1'];
333         $encrypted_email = $_GET['u2'];
334         $email = decryptDataString($encrypted_email);
335         $time_out = $_GET['u3'];
336         $encrypted_secret_key = getRegistrationtKey($email, $time_out);
337         $now = time();
338         if ($now > $time_out) {
339           $message = Yii::t("story", "We are sorry, this link has expired. You will need to sign up again.");
340         } else if ($secret_key != $encrypted_secret_key) {
341           $message = Yii::t("story", "This is not a valid link.");
342         } else {
343           //update in identity manager
344           $module = Yii::app()->getModule('backendconnector');
345           if (empty($module)) {
346             throw new Exception(Yii::t('story', 'backendconnector module is missing or not defined'));
347           }
348           $userIdentityApi = new UserIdentityAPI();
349           $userInfo = $userIdentityApi->getUserDetail(IDM_USER_ENTITY, array('email' => trim($email)), false, true);
350           $userId = '';
351           if (array_key_exists('_items', $userInfo) && array_key_exists(0, $userInfo['_items']) && array_key_exists('_id', $userInfo['_items'][0])) {
352             $userId = $userInfo['_items'][0]['_id'];
353           }
354           if (empty($userId)) {
355             throw new Exception('User id is empty for email ' . $email);
356           }
357           $inputParam = array(
358               'status' => '1',
359               'id' => $userId
360           );
361           $updateUser = $userIdentityApi->curlPut(IDM_USER_ENTITY, $inputParam);
362           if (array_key_exists('_status', $updateUser) && $updateUser['_status'] == 'OK') {
363             $verified = TRUE;
364             $message = Yii::t("story", "Your account has been activated. Please login");
365           } else {
366             throw new Exception('Failed to update status as active for email ' . $email);
367           }
368         }
369       } else {
370         $message = Yii::t("story", "We are sorry, the page you are looking for seems to be missing.");
371       }
372     } catch (Exception $e) {
373       Yii::log($e->getMessage(), ERROR, 'Error in actionActivateUser');
374       $message = Yii::t('story', 'Some technical problem occurred, contact administrator');
375     }
376     $this->render('activation', array('message' => $message, 'verified' => $verified));
377   }
