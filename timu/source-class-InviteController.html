 44   /**
 45    * Action used for listing all existing user and invited story
 46    * @param $_GET['slug'] story slug
 47    * @return null
 48    */
 49   public function actionList() {
 50     includeJsFile(array('position_end' => array('invite.js')));
 51     try {
 52       if (!array_key_exists('slug', $_GET) || empty($_GET['slug'])) {
 53         $this->redirect(BASE_URL);
 54       }
 55       $invitedUser = array();
 56       $story = new Story();
 57       $story->slug = $_GET['slug'];
 58       $storyDetail = $story->get();
 59       if (empty($storyDetail)) {
 60         throw new Exception(Yii::t('story', 'story slug is not valid'));
 61       }
 62       if ($storyDetail['type'] != INVITE_STORY_TYPE) {
 63         Yii::log('Info from invite user listing', INFO, 'Story type is not invite');
 64         $this->redirect(BASE_URL);
 65       }
 66       $invite = new Invite();
 67       $invite->storyId = $storyDetail['id'];
 68       $invitedUsers = $invite->get();
 69       if (!empty($invitedUsers)) {
 70         foreach ($invitedUsers as $user) {
 71           $invitedUser[] = array(
 72             'user_email' => $user['invitation_to'],
 73             'invitation_date' => $user['invite_date'],
 74             'status' => $user['status'],
 75           );
 76         }
 77       }
 78     } catch (Exception $e) {
 79       Yii::log('Error in invite user listing', ERROR, $e->getMessage());
 80     }
 81     $this->render('list', array('invited_users' => $invitedUser, 'story' => $storyDetail));
 82   }

 84 /**
 85  * Function  used for listing all invited user
 86  * @param $_GET['slug'] story slug
 87  * @return [type] [description]
 88  */
 89   public function actionInvitation() {
 90     try {
 91       $story = new Story();
 92       if (!isset(Yii::app()->session['user'])) {
 93         $this->redirect(BASE_URL);
 94       }
 95       $story->slug = $_GET['slug'];
 96       $storyDetail = $story->get();
 97       if (empty($storyDetail)) {
 98         $this->redirect(BASE_URL . 'admin/invitation/list/' . $_GET['slug']);
 99       }
100       if (!empty($_POST)) {
101         $postData = array_map('trim', $_POST);
102         if (!array_key_exists('email', $postData) || empty($postData['email'])) {
103           throw new Exception(Yii::t('story', 'Email id can not be empty'));
104         }
105         $mail = new EMail();
106         $mail->to = array($postData['email']);
107         $mail->subject = Yii::t('story', 'Invitation for adding story');
108         $mail->from = Yii::app()->session['user']['email'];
109         $mail->fromName = Yii::app()->session['user']['firstname']. ' ' .
110                           Yii::app()->session['user']['lastname'];
111         $mail->body = $this->prepareMailBody($storyDetail['title']);
112         if (!$mail->sendMail()) {
113           throw new Exception('Failed to sending mail');
114         }
115         $this->saveInvitation($postData['email'], $storyDetail['id']);
116       }
117     } catch (Exception $e) {
118       Yii::log('Error in actionInvitation', ERROR, $e->getMessage());
119     }
120     $this->redirect(BASE_URL . 'admin/invitation/list/' . $_GET['slug']);
121   }

125  * @param  string $storyTitle Title of story
126  * @return string html body

128   public function prepareMailBody($storyTitle) {
129     $html = '';
130     $html = file_get_contents(Yii::app()->theme->basePath . '/views/invite/invitationMail.html');
131     $html = str_replace("{{invitation_text}}", Yii::t('story', 'Invitation'), $html);
132     $html = str_replace("{{invite_text}}", Yii::t('story', 'You are cordially invited to add your story on'), $html);
133     $html = str_replace("{{story_title}}", $storyTitle, $html);
134     $html = str_replace("{{registration_url}}", REGISTRATION_URL, $html);
135     $html = str_replace("{{register_link_text}}", Yii::t('story', 'Click here for registration'), $html);
136     $html = str_replace("{{story_url}}", BASE_URL.'story/' .$_GET['slug'], $html);
137     $html = str_replace("{{already_register_text}}", Yii::t('story', 'If you are already registered then'), $html);
138     $html = str_replace("{{add_story_text}}", Yii::t('story', 'add story'), $html);
139     return $html;

168   public function updateInvitedUser($storyId) {
169     try {
170       if (empty($storyId)) {
171         return false;
172       }
173       $updateStatus = false;
174       $invite = new Invite();
175       $invite->storyId = $storyId;
176       $invite->invitationTo = Yii::app()->session['user']['email'];
177       $invite->status = APPROVED;
178       $isUpdate = $invite->update();
179       if (is_numeric($isUpdate)) {
180         $updateStatus = true;
181       } else {
182         Yii::log('Error in updateInvitedUser', ERROR, $isUpdate);
183       }
184     } catch (Exception $e) {
185       Yii::log('Error in updateInvitedUser', ERROR, $e->getMessage());
186     }
