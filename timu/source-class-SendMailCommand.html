16   public function run($args) {
17     try {
18       if (!array_key_exists(0, $args) || empty($args[0])) {
19         throw new Exception('Subject can not be empty');
20       }
21       if (!array_key_exists(1, $args) || empty($args[1])) {
22         throw new Exception('Body can not be empty');
23       }
24       $subject = $args[0];
25       $subject = str_replace("$3#$", "'", $subject);
26       $subject = html_entity_decode($subject);
27       $body = $args[1];
28       $body = str_replace("$3#$", "'", $body);
29       $email = new EMail();
30       $email->subject = $subject;
31       $email->from = 'noreply@civiclinks.it';
32       if (defined('DEFAULT_EMAIL')) {
33         $email->from = DEFAULT_EMAIL;
34       }
35       $email->body = htmlspecialchars_decode($body);
36       $moderatorEmails = '';
37       if (array_key_exists(2, $args) && (($args[2] == 'registeration_activation_mail')
38         || $args[2] == 'forgot_password_mail')) {
39         if (!array_key_exists(3, $args) || empty($args[3])) {
40           throw new Exception('Email can not be empty');
41         }
42         $moderatorEmails = $args[3];
43       }
44       if (empty($moderatorEmails)) {
45         $config = new Configuration();
46         $config->type = 'config';
47         $configurations = $config->get();
48         foreach ($configurations as $confiuration) {
49           if ($confiuration['name_key'] == 'moderators_email') {
50             $moderatorEmails = $confiuration['value'];
51           }
52         }
53       }
54       if (!empty($moderatorEmails)) {
55         $moderatorsEmail = explode(',', $moderatorEmails);
56         foreach ($moderatorsEmail as $emailId) {
57           $email->to = array($emailId);
58           $emailResponse = $email->sendMail();
59           if (array_key_exists('success', $emailResponse) && $emailResponse['success'] == false) {
60             Yii::log('Failed to send mail :' . $emailId, 'error', 'Error in sending mail command');
61           }
62         }
63       }
64     } catch (Exception $e) {
65       Yii::log($e->getMessage(), 'error', 'Error in SendMailCommand');
66     }
67   }
