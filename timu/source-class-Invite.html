102   public function update() {
103     $connection = Yii::app()->db;
104     if (empty($this->storyId)) {
105       throw new Exception(Yii::t('story', 'story id can not be empty'));

107     if (empty($this->status)) {
108       throw new Exception(Yii::t('story', 'Invited author status can not be empty'));
109     }
110     if (empty($this->invitationTo)) {
111       throw new Exception(Yii::t('story', 'Invited author email can not be empty'));
112     }
113     $sql = "UPDATE invitation SET status = :status WHERE story_id = :story_id AND
114       invitation_to = :invitation_to";
115      $query = $connection->createCommand($sql);
116     $query->bindParam(":story_id", $this->storyId);
117     $query->bindParam(":status", $this->status);
118     $query->bindParam(":invitation_to", $this->invitationTo);
119     $isUpdate = $query->execute();
120     // if  $isUpdate is numeric it will return 1 else return error
121     if (is_numeric($isUpdate)) {
