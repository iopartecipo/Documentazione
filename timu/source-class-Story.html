 35    * This function is used to get all stories.

 38   public function get() {
 39     $connection = Yii::app()->db;
 40     $where = array(1);
 41     $data = array();
 42     if (!empty($this->slug)) {
 43       $where[] = 'slug = :slug';
 44       $data[':slug'] = $this->slug;
 45     }
 46     if (isset($this->status)) {
 47       $where[] = 'status = :status';
 48       $data[':status'] = $this->status;
 49     }
 50     if (!empty($this->projectTag)) {
 51       $projectTags = explode(',', $this->projectTag);
 52       foreach ($projectTags as $key => $tag) {
 53         $tagCondition[] = 'project_tag like :project_tag' . $key;
 54         $data[ ':project_tag' . $key] = '%' . trim($tag) . '%';
 55       }
 56       $where[] = ' ( ' . implode(' || ', $tagCondition) . ' ) ';
 57     }
 58     $limit = '';
 59     if (!empty($this->limit)) {
 60       $limit = ' LIMIT 0,' . $this->limit;
 61     }
 62     $sql = "SELECT id, title, description, type, image, logo, slug, status, creation_time, 
 63       project_tag FROM story WHERE " . implode(' AND ', $where) . $limit;
 64     $query = $connection->createCommand($sql);
 65     foreach ($data as $key => &$value) {
 66       $query->bindParam($key, $value);
 67     }
 68     if (!empty($this->slug)) {
 69       return $query->queryRow();
 70     }
 71     return $query->queryAll();
 72   }

190    * This function is used to submit Post related to a Story.
191    * @param int $storyId
192    * @param string $storySlug

194   public function submitPost($storyId) {
195     try {
196       $aggregator = new AggregatorManager();
197       $aggregator->id = $storyId;
198       $aggregator->description = htmlspecialchars($this->description);
199       $aggregator->authorName = Yii::app()->session['user']['firstname'].' '.Yii::app()->session['user']['lastname'];
200       $aggregator->authorSlug = Yii::app()->session['user']['id'];
201       $aggregator->source = SOURCE;
202       $aggregator->latitude = $this->latitude;
203       $aggregator->longitude = $this->longitude;
204       $aggregator->tags = $this->tags;
205 
206       // Fetching all the image src tag from the html of description.
207       $allowedFiles = array();
208       $images = array();
209       $allAllowedFiles = array();
210       if (defined('ALLOWED_FILE_EXTENTIONS') == FALSE) {
211         throw new Exception(Yii::t('story', 'Constant for Allowed files extentions is not defined'));
212       }
213       $allowedFileExtentions = json_decode(ALLOWED_FILE_EXTENTIONS, TRUE);
214       preg_match_all( '|<img.*?src=[\'"](.*?)[\'"].*?>|i', $this->description, $images, PREG_SET_ORDER);
215       preg_match_all( '|<iframe.*?src=[\'"](.*?)[\'"].*?>|i', $this->description, $allowedFiles, PREG_SET_ORDER);
216       if (!empty($allowedFiles)) {
217         foreach ($allowedFiles as $file) {
218           if (array_key_exists(1, $file) && !empty($file[1])) {
219             $fileExtention = explode('.', $file[1]);
220             $fileExtention = end($fileExtention);
221             if (!empty($allowedFileExtentions) && in_array($fileExtention, $allowedFileExtentions)) {
222               $allAllowedFiles[$fileExtention][] = $file[1];
223             } else {
224               $pattern = array();
225               preg_match('/^.*(player.|www.)?(vimeo\.com|youtu(be\.com|\.be))\/(video\/|embed\/|watch\?v=)?([A-Za-z0-9._%-]*)(\&\S+)?/', $file[1], $pattern);
226               if (isset($pattern[2]) && ($pattern[2] == 'youtube.com' || $pattern[2] == 'vimeo.com')) {
227                 if (isset($pattern[5])) {
228                   $allAllowedFiles['video'][] = 'http:' . $file[1];
229                 }
230               }
231             }
232           }
233         }
234       }
235       $response = $aggregator->savePost($images, $allAllowedFiles);
236       if ($response['success']) {
237         $response['msg'] = Yii::t('story', 'You have successfully submitted a Post for a Story');
238       } else {
239         $response['msg'] = Yii::t('story', 'Some technical problem occurred, contact administrator');
240       }
241     } catch (Exception $e) {
242       Yii::log('', ERROR, Yii::t('story', 'Error in submitPost :') . $e->getMessage());
243       $response['success'] = false;
244       $response['msg'] = $e->getMessage();
245     }
246     return $response;
247   }

264       $posts = $aggregatorManager->getEntry(POST_LIMIT, $this->offset, '', 'active', '', '', '', '', '', '', '', '', array(), $this->sort, 'status,title,author,id,content,links', '', '', trim('story,' . $storyId), SOURCE);
265     } else {
266       $posts = $aggregatorManager->getEntry(POST_LIMIT, '', '', 'active', $this->tags, '', '', $this->count, '', '', '', '', array(), $this->sort, '*', '', '', trim('story,' . $storyId), SOURCE);
