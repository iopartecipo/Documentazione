20   public function run($args) {
21     try {
22       $compressFileName = '';
23       if (array_key_exists(0, $args) && !empty($args[0])) {
24         $compressFileName = $args[0];
25       } else {
26         $files = glob('*.tar.gz');
27         $fileCreationTime = 0;
28         foreach ($files as $file) {
29           if (filectime($file) > $fileCreationTime) {
30             $fileCreationTime = filectime($file);
31             $compressFileName = $file;
32           }
33         }
34       }
35       if (!file_exists($compressFileName)) {
36         throw new Exception( 'file does not exist - ' . $compressFileName);
37       }
38       $finfoObject = finfo_open(FILEINFO_MIME_TYPE);
39       if (strpos(finfo_file($finfoObject, $compressFileName), 'gzip') === FALSE) {
40         throw new Exception('Only tar.gz file type is allowed - ' . $compressFileName);
41       };
42       $fileInfo = pathinfo($compressFileName);
43       if (array_key_exists('filename', $fileInfo) && file_exists($fileInfo['filename'])) {
44         unlink($fileInfo['filename']);
45       }
46 
47       // decompress from gzip
48       $decompressFile = new PharData($compressFileName);
49       $decompressFile->decompress();
50       if (!array_key_exists('filename', $fileInfo) || !file_exists($fileInfo['filename'])) {
51         throw new Exception('failed to decompress file - ' . $compressFileName);
52       }
53       //extract tar
54       $phar = new PharData($fileInfo['filename']);
55       $phar->extractTo(realpath(dirname(__FILE__) . '/../../'), null, true);
56       unlink($fileInfo['filename']);
57     } catch (Exception $e) {
58       print_r($e->getMessage()."\n");

60     }
