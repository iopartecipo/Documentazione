 18   /**
 19    * detectAndTranslateLanguage
 20    * function is used for detaect language and message translatation
 21    * @param  $message - array of string
 22    * @param  $targetLang - language for translation
 23    * @return array $resp - detected source language and translation 
 24    * @author Sankalp Mishra<sankalp@incaendo.com>
 25    */
 26   public function detectAndTranslateLanguage() {
 27     try {
 28       $resp = array();
 29       if (empty($this->message)) {
 30         return $resp;
 31       }
 32       if (is_array($this->message)) {
 33         $strToBeTranslate = '';
 34         foreach ($this->message as $msg) {
 35           $strToBeTranslate .= '&q=' . urlencode($msg);
 36         }
 37       } else {
 38         $strToBeTranslate = '&q=' . urlencode($this->message);;
 39       }
 40       $url = 'https://www.googleapis.com/language/translate/v2?key=' . TRANSLATION_API_KEY . '&target=' . $this->targetLang . $strToBeTranslate;
 41       $defaultParams = array(CURLOPT_URL => $url,
 42           CURLOPT_RETURNTRANSFER => 1,
 43           CURLOPT_HEADER => 0,
 44           CURLOPT_TIMEOUT => CURL_TIMEOUT);
 45       $curlHandle = curl_init();
 46       curl_setopt_array($curlHandle, $defaultParams);
 47       $translateTxt = json_decode(curl_exec($curlHandle));
 48       $headers = curl_getinfo($curlHandle);
 49       curl_close($curlHandle);
 50       if ($headers['http_code'] != '200') {
 51         throw new Exception(Yii::t('discussion', 'Google api for translation is not respondind'));
 52       }
 53       if (!empty($translateTxt->data->translations)) {
 54         $resp['source_language'] = $translateTxt->data->translations[0]->detectedSourceLanguage;
 55         foreach ($translateTxt->data->translations as $text) {
 56           $resp['translated_text'][] = $text;
 57         }
 58       }
 59       return $resp;
 60     } catch (Exception $e) {
 61       Yii::log('Error in languageDetectAndTranslator : ', ERROR, $e->getMessage());
 62     }
 63   }

 80   public function saveTranslatedProposal($id, $translatedLang) {
 81     $resp = array('success' => false, 'msg' => '');
 82     if (empty($id)) {
 83       $resp['msg'] = Yii::t('discussion', 'Proposal id is empty');
 84       return $resp;
 85     }
 86     if ($translatedLang == 'sl') {
 87       $nextTranslation = 'italian';
 88     } else if ($translatedLang == 'it') {
 89       $nextTranslation = 'slovenian';
 90     }
 91     $resp['msg']['nextTranslation'] = $nextTranslation;
 92     $aggregatorManager = new AggregatorManager();
 93     $related = trim('proposal-translated,' . $id);
 94     $proposal = $aggregatorManager->getEntry('', '', '', 'active', $translatedLang, '', '', 0, '', '', '', '', array(), '', 'status,title,content,author,tags', '', '', $related, '', '');
 95     if (!empty($proposal)) {
 96       $resp['msg']['title'] = $proposal[0]['title'];
 97       $resp['msg']['description'] = preg_replace('#<br\s*/?>#i', "\n", $proposal[0]['content']['description']);
 98       $resp['msg']['summary'] = preg_replace('#<br\s*/?>#i', "\n", $proposal[0]['content']['summary']);
 99     } else {
100       $proposal = $aggregatorManager->getEntry('', '', $id, 'active', '', '', '', 0, '', '', '', '', array(), '', 'status,title,content,author,tags', '', '', '', '', '');
101       $proposal = $proposal[0];
102       $str = array();
103       if (array_key_exists('title', $proposal) && !empty($proposal['title'])) {
104         $str[] = $proposal['title'];
105       }
106       if (array_key_exists('content', $proposal) && !empty($proposal['content'])) {
107         if (array_key_exists('description', $proposal['content']) && !empty($proposal['content']['description'])) {
108           $str[] = $proposal['content']['description'];
109         }
110         if (array_key_exists('summary', $proposal['content']) && !empty($proposal['content']['summary'])) {
111           $str[] = $proposal['content']['summary'];
112         }
113       }
114       $translatedStr = $this->languageDetectAndTranslator($str, $translatedLang);
115       if (array_key_exists('translated_text', $translatedStr)) {
116         if (array_key_exists(0, $translatedStr['translated_text']) && array_key_exists('translatedText', $translatedStr['translated_text'][0])) {
117           $aggregatorManager->title = $translatedStr['translated_text'][0]->translatedText;
118           $resp['msg']['title'] = $translatedStr['translated_text'][0]->translatedText;
119         }
120         if (array_key_exists(1, $translatedStr['translated_text']) && array_key_exists('translatedText', $translatedStr['translated_text'][0])) {
121           $aggregatorManager->description = $translatedStr['translated_text'][1]->translatedText;
122           $resp['msg']['description'] = preg_replace('#<br\s*/?>#i', "\n", $translatedStr['translated_text'][1]->translatedText);
123         }
124         if (array_key_exists(2, $translatedStr['translated_text']) && array_key_exists('translatedText', $translatedStr['translated_text'][0])) {
125           $aggregatorManager->summary = $translatedStr['translated_text'][2]->translatedText;
126           $resp['msg']['summary'] = preg_replace('#<br\s*/?>#i', "\n", $translatedStr['translated_text'][2]->translatedText);
127         }
128       }
129       if (array_key_exists('author', $proposal)) {
130         $aggregatorManager->authorName = $proposal['author']['name'];
131         $aggregatorManager->authorSlug = $proposal['author']['slug'];
132       }
133       $aggregatorManager->relationType = 'proposal-translated';
134       $aggregatorManager->relatedTo = $id;
135       foreach ($proposal['tags'] as $tag) {
136         if ($tag['scheme'] == LANGUAGE_SCHEME) {
137           $tag['name'] = $translatedLang;
138           $tag['slug'] = strtolower($translatedLang);
139         }
140         $tags[] = $tag;
141       }
142       $aggregatorManager->tags = $tags;
143       $aggregatorManager->saveProposal();
144     }
145     return $resp;
146   }
