46   private function _prepareProposalTags($proposal, $opinions) {
47     if (array_key_exists('tags', $proposal)) {
48       $triangle = array();
49       $opinionCount = 0;
50       foreach ($opinions as $opinion) {
51         if (array_key_exists('tags', $opinion)) {
52           foreach ($opinion['tags'] as $tag) {
53             if ($tag['scheme'] == INDEX_TAG_SCHEME) {
54               if (array_key_exists($tag['weight'], $triangle)) {
55                 $triangle[$tag['weight']] += 1;
56               } else {
57                 $triangle[$tag['weight']] = 1;
58               }
59             }
60           }
61         }
62         if (array_key_exists('count', $opinion)) {
63           $opinionCount = $opinion['count'];
64         }
65       }
66       $countExists = 0;
67       foreach ($proposal['tags'] as &$tag) {
68         if ($tag['scheme'] == OPINION_COUNT_TAG_SCEME) {
69           $tag['weight'] = $opinionCount;
70           $countExists = 1;
71         }
72         if ($tag['scheme'] == TAG_SCHEME) {
73           if (array_key_exists($tag['name'], $triangle)) {
74             $tag['weight'] = $triangle[$tag['name']];
75           } else {
76             $tag['weight'] = 0;
77           }
78         }
79       }
80       if ($countExists == 0) {
81         $proposal['tags'][] = array('name' => 'OpinionCount', 'scheme' => OPINION_COUNT_TAG_SCEME,
82                                     'slug' => 'OpinionCount', 'weight' => $opinionCount);
83       }
84       return $proposal['tags'];
85     }
